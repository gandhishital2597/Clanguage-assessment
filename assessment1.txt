●  Write a program to make multiplication of 2-D Matrix
#include <stdio.h>

int main() {
int a[10][10], b[10][10], c[10][10];
int r1, c1, r2, c2;
int i, j, k;


printf("Enter rows and columns for first matrix: ");
scanf("%d %d", &r1, &c1);

printf("Enter rows and columns for second matrix: ");
scanf("%d %d", &r2, &c2);


if (c1 != r2) {
printf("Matrix multiplication not possible.\n");
return 0;
}

// Input elements of first matrix
printf("\n----------------Matrix 1:-----------------------\n");
printf("Enter elements of first matrix:\n");
for (i = 0; i < r1; i++) {
for (j = 0; j < c1; j++) {
printf("Enter elements : ");
scanf("%d", &a[i][j]);
}
}

// Input elements of second matrix
printf("\n----------------Matrix 2:-----------------------\n");
printf("Enter elements of second matrix:\n");
for (i = 0; i < r2; i++) {
for (j = 0; j < c2; j++) {
printf("Enter elements : ");
scanf("%d", &b[i][j]);
}
}

// Initialize result matrix elements to 0
for (i = 0; i < r1; i++) {
for (j = 0; j < c2; j++) {
c[i][j] = 0;
}
}

// Multiply matrices
for (i = 0; i < r1; i++) {
for (j = 0; j < c2; j++) {
for (k = 0; k < c1; k++) {
c[i][j] += a[i][k] * b[k][j];
}
}
}

// Display the result
printf("Result : Multiplication matrix:\n");
for (i = 0; i < r1; i++) {
for (j = 0; j < c2; j++) {
printf("%d\t", c[i][j]);
}
printf("\n");
}

return 0;
}
