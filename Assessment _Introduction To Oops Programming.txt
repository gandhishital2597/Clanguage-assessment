● Create a lecture management System
#include <iostream>
#include <string>
using namespace std;

// Class to represent Lecture Details
class Lecture {
private:
// Data members
string lecturerName;
string subjectName;
string courseName;
int numberOfLectures;

public:
// Default constructor
Lecture() {
lecturerName = "";
subjectName = "";
courseName = "";
numberOfLectures = 0;
}

// Parameterized constructor to assign initial values
Lecture(string lName, string sName, string cName, int nLectures) {
lecturerName = lName;
subjectName = sName;
courseName = cName;
numberOfLectures = nLectures;
}

// Function to assign or update lecture details
void addLectureDetails() {
cout << "\nEnter Lecturer Name: ";
cin.ignore();
getline(cin, lecturerName);

cout << "Enter Subject Name: ";
getline(cin, subjectName);

cout << "Enter Course Name: ";
getline(cin, courseName);

cout << "Enter Number of Lectures: ";
cin >> numberOfLectures;
}

// Function to display lecture details
void displayLectureDetails() {
cout << "\n--- Lecture Details ---" << endl;
cout << "Lecturer Name: " << lecturerName << endl;
cout << "Subject Name: " << subjectName << endl;
cout << "Course Name: " << courseName << endl;
cout << "Number of Lectures: " << numberOfLectures << endl;
}
};

int main() {
const int MAX_LECTURERS = 5; // Program handles at least 5 lecturers
Lecture lecturers[MAX_LECTURERS]; // Array of Lecture objects

int choice;
int count = 0; // Tracks number of lecturers added

do {
cout << "\n=== Lecture Management System ===" << endl;
cout << "1. Add Lecture Details" << endl;
cout << "2. Display Lecture Details" << endl;
cout << "3. Exit" << endl;
cout << "Enter your choice: ";
cin >> choice;

switch (choice) {
case 1:
if (count < MAX_LECTURERS) {
lecturers[count].addLectureDetails();
count++;
} else {
cout << "Maximum number of lecturers reached!" << endl;
}
break;

case 2:
if (count == 0) {
cout << "No lecture details added yet!" << endl;
} else {
for (int i = 0; i < count; i++) {
cout << "\nLecturer #" << (i + 1) << endl;
lecturers[i].displayLectureDetails();
}
}
break;

case 3:
cout << "Exiting the program. Goodbye!" << endl;
break;

default:
cout << "Invalid choice! Please try again." << endl;
}
} while (choice != 3);

return 0;
}
